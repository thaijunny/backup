    @Override

        try {

                    String rawRegPeriod = request.get(i).getRegPeriod();

                        regPeriodDate = LocalDate.parse(rawRegPeriod, inputFormatter).format(outputFormatter);
                    }
                    LoanRegistrationExtraRequest vehInspection = request.get(i);
                    businessStatus = businessStatusRepository.createVehInspection(
                            regPeriodDate,
                            vehInspection.getRegLoad(),

    @Override
    public List<InformationPropertyItemDTO> getInformationProperty(InformationPropertyRequest request) {
        List<InformationPropertyResponse> entityList =
                loanRegDetailRepository.getInformationProperty(request.getLoanRegistrationId());
        Map<Integer, List<InformationPropertyResponse>> groupedByCar =
                entityList.stream().collect(Collectors.groupingBy(InformationPropertyResponse::getLoanregcarid));

        List<InformationPropertyItemDTO> dtoList = new ArrayList<>();

        for (Map.Entry<Integer, List<InformationPropertyResponse>> entry : groupedByCar.entrySet()) {
            List<InformationPropertyResponse> carData = entry.getValue();
            InformationPropertyItemDTO dto = new InformationPropertyItemDTO();
            BeanUtils.copyProperties(carData.get(0), dto);

            List<SimilarAssetsBusiness> assets = carData.stream()
                    // check nếu link website null hoặc value = 0 thì sẽ trả ra tài sản tương tự mảng rỗng
                    .filter(e -> e.getLinkwebsite() != null && e.getValue() != null && e.getValue().compareTo(BigDecimal.ZERO) > 0)
                    .map(e -> {
                        SimilarAssetsBusiness asset = new SimilarAssetsBusiness();
                        asset.setId(e.getId() != null ? e.getId() : 0);
                        asset.setLoanRegCarId(e.getLoanregcarid() != null ? e.getLoanregcarid() : 0);
                        asset.setValue(e.getValue() != null ? e.getValue() : BigDecimal.ZERO);
                        asset.setLinkWebsite(e.getLinkwebsite());
                        return asset;
                    }).collect(Collectors.toList());

            dto.setSimilarAssetsBusinesses(assets);
            dtoList.add(dto);
        }
        return dtoList;
    }

                        businessStatus = businessStatusRepository.createSimilarAssetsBusiness(
                                request.getLoanCarId(),
                                similarAssetsBusinessRequest.getValue(),
                                similarAssetsBusinessRequest.getLinkWebsite());
                    }
                }
            }

        } catch (Exception e) {
            throw new BusinessException(ErrorCode.UNKNOWN_ERROR, e.getMessage());
        }
        return businessStatus;
    }

    private void validateBigDecimalNotBlank(BigDecimal value,
                                            String errorCode,
                                            String description,
                                            String errorCode2,String description2
    ) {
        //  Check null
        if (value == null) {
            throw new BusinessException(errorCode, description);
        }
        if (value.compareTo(BigDecimal.ZERO) < 0) {
            throw new BusinessException(ErrorCode.LOAN_AMOUNT_NEGATIVE, ErrorCode.LOAN_AMOUNT_NEGATIVE_DESCRIPTION);
        }
    }
    private void validateBigDecimal(BigDecimal value) {
        //  Check null
        if (value == null) {
            throw new BusinessException(ErrorCode.LOAN_AMOUNT_BLANK, ErrorCode.LOAN_AMOUNT_BLANK_DESCRIPTION);
        }

        //  Check âm
        if (value.compareTo(BigDecimal.ZERO) < 0) {
            throw new BusinessException(ErrorCode.LOAN_AMOUNT_NEGATIVE, ErrorCode.LOAN_AMOUNT_NEGATIVE_DESCRIPTION);
        }

        // Check > 3 tỷ
        BigDecimal max = BigDecimal.valueOf(3_000_000_000L);
        if (value.compareTo(max) > 0) {
            throw new BusinessException(ErrorCode.LOAN_AMOUNT_TOO_LARGE, ErrorCode.LOAN_AMOUNT_TOO_LARGE_DESCRIPTION);
        }
        if (value.compareTo(BigDecimal.ZERO) == 0) {
            throw new BusinessException(ErrorCode.LOAN_AMOUNT_FULL_ZERO, ErrorCode.LOAN_AMOUNT_FULL_ZERO_DESCRIPTION);
        }
    }

        validateNotBlank(request.getGuardMobile(), ErrorCode.GUARDIAN_MOBILE_BLANK, ErrorCode.GUARDIAN_MOBILE_BLANK_DESCRIPTION);
        validateNotBlank(request.getGuardBirth(), ErrorCode.GUARDIAN_BIRTHDAY_BLANK, ErrorCode.GUARDIAN_BIRTHDAY_BLANK_DESCRIPTION);
        validateNotBlank(request.getGuardMaritalStatus(), ErrorCode.GUARDIAN_MARITAL_BLANK, ErrorCode.GUARDIAN_MARITAL_BLANK_DESCRIPTION);
        validateNotBlank(request.getGuardIdNumber(), ErrorCode.GUARDIAN_ID_NUMBER_BLANK, ErrorCode.GUARDIAN_ID_NUMBER_BLANK_DESCRIPTION);
        validateNotBlank(request.getGuardIdDate(), ErrorCode.GUARDIAN_ID_DATE_BLANK, ErrorCode.GUARDIAN_ID_DATE_BLANK_DESCRIPTION);
        validateNotBlank(request.getGuardIdPlace(), ErrorCode.GUARDIAN_ID_PLACE_BLANK, ErrorCode.GUARDIAN_ID_PLACE_BLANK_DESCRIPTION);
        validateNotBlank(request.getGuardCurrentAddress(), ErrorCode.GUARDIAN_CURRENT_ADDRESS_BLANK, ErrorCode.GUARDIAN_CURRENT_ADDRESS_BLANK_DESCRIPTION);
        validateNotBlank(request.getGuardAddress(), ErrorCode.GUARDIAN_ADDRESS_BLANK, ErrorCode.GUARDIAN_ADDRESS_BLANK_DESCRIPTION);
        validateNotBlankInteger(request.getGuardGender(), ErrorCode.GUARDIAN_GENDER_BLANK, ErrorCode.GUARDIAN_GENDER_BLANK_DESCRIPTION);
        validateNotBlankInteger(request.getGuardWardId(), ErrorCode.GUARDIAN_WARD_ID_BLANK, ErrorCode.GUARDIAN_WARD_ID_BLANK_DESCRIPTION);
//        validateNotBlankInteger(request.getGuardDistrictId(), ErrorCode.GUARDIAN_DISTRICT_ID_BLANK, ErrorCode.GUARDIAN_DISTRICT_ID_BLANK_DESCRIPTION);
        validateNotBlankInteger(request.getGuardProvinceId(), ErrorCode.GUARDIAN_PROVINCE_ID_BLANK, ErrorCode.GUARDIAN_PROVINCE_ID_BLANK_DESCRIPTION);
        validateNotBlankInteger(request.getGuardResidenceWardId(), ErrorCode.GUARDIAN_RESIDENCE_WARD_ID_BLANK, ErrorCode.GUARDIAN_RESIDENCE_WARD_ID_BLANK_DESCRIPTION);
//        validateNotBlankInteger(request.getGuardResidenceDistrictId(), ErrorCode.GUARDIAN_RESIDENCE_DISTRICT_ID_BLANK, ErrorCode.GUARDIAN_RESIDENCE_DISTRICT_ID_BLANK_DESCRIPTION);
        validateNotBlankInteger(request.getGuardResidenceProvinceId(), ErrorCode.GUARDIAN_RESIDENCE_PROVINCE_ID_BLANK, ErrorCode.GUARDIAN_RESIDENCE_PROVINCE_ID_BLANK_DESCRIPTION);
        validateNotBlankInteger(request.getGuardLoanStatusFiny(), ErrorCode.GUARDIAN_LOAN_STATUS_FINY_BLANK, ErrorCode.GUARDIAN_LOAN_STATUS_FINY_BLANK_DESCRIPTION);

        DateValidator.validateDateFormatAndExistence(request.getGuardIdDate(), ErrorCode.GUARDIAN_ID_DATE_FORMAT, ErrorCode.GUARDIAN_ID_DATE_FORMAT_DESCRIPTION, ErrorCode.GUARDIAN_ID_DATE_FORMAT_IS_REAL, ErrorCode.GUARDIAN_ID_DATE_FORMAT_IS_REAL_DESCRIPTION);
        DateValidator.validateDateFormatAndExistence(request.getGuardBirth(), ErrorCode.GUARDIAN_BIRTHDAY_FORMAT, ErrorCode.GUARDIAN_BIRTHDAY_FORMAT_DESCRIPTION, ErrorCode.GUARDIAN_BIRTHDAY_FORMAT_IS_REAL, ErrorCode.GUARDIAN_BIRTHDAY_FORMAT_IS_REAL_DESCRIPTION);
        DateValidator.validateRealDate(request.getGuardIdDate(), ErrorCode.GUARDIAN_ID_DATE_REAL, ErrorCode.GUARDIAN_ID_DATE_REAL_DESCRIPTION);

        DateValidator.validateRealDateNow(request.getGuardIdDate(), ErrorCode.GUARDIAN_ID_DATE_REAL_NOW, ErrorCode.GUARDIAN_ID_DATE_REAL_DESCRIPTION_NOW);
        DateValidator.validateDateBeforeNow(request.getGuardIdDate(), ErrorCode.GUARDIAN_ID_DATE_REAL_NOW, ErrorCode.GUARDIAN_ID_DATE_REAL_DESCRIPTION_NOW);


        DateValidator.validateRealDate(request.getGuardBirth(), ErrorCode.GUARDIAN_DOB_REAL, ErrorCode.GUARDIAN_DOB_REAL_DESCRIPTION);
        DateValidator.validateDateBeforeNow(request.getGuardBirth(), ErrorCode.GUARDIAN_DOB_FUTURE, ErrorCode.GUARDIAN_DOB_FUTURE_DESCRIPTION);
        validateSpace(request.getGuardMobile(), ErrorCode.GUARDIAN_MOBILE_SPACE, ErrorCode.GUARDIAN_MOBILE_SPACE_DESCRIPTION);
        validateSpace(request.getGuardIdNumber(), ErrorCode.GUARDIAN_ID_NUMBER_SPACE, ErrorCode.GUARDIAN_ID_NUMBER_SPACE_DESCRIPTION);
        validateSpace(request.getGuardName(), ErrorCode.GUARDIAN_FULL_NAME_SPACE, ErrorCode.GUARDIAN_FULL_NAME_SPACE_DESCRIPTION);

        if (!request.getGuardMobile().matches("^\\d{10}$")) {
            throw new BusinessException(ErrorCode.GUARDIAN_MOBILE_INVALID, ErrorCode.GUARDIAN_MOBILE_INVALID_DESCRIPTION);
        }

            throw new BusinessException(ErrorCode.GUARDIAN_ID_NUMBER_VALID, ErrorCode.GUARDIAN_ID_NUMBER_VALID_DESCRIPTION);
        }

        if (request.getGuardName().trim().length() > 100 || !request.getGuardName().matches("^[\\p{L} ]+$")) {
            throw new BusinessException(ErrorCode.GUARDIAN_FULL_NAME_VALID, ErrorCode.GUARDIAN_FULL_NAME_VALID_DESCRIPTION);
        }

        validateOneHundredCharacter(request.getGuardAddress(), ErrorCode.GUARD_ADDRESS_TOO_LONG, ErrorCode.GUARD_ADDRESS_TOO_LONG_DESCRIPTION);
        validateOneHundredCharacter(request.getGuardCurrentAddress(), ErrorCode.GUARD_CURRENT_ADDRESS_TOO_LONG, ErrorCode.GUARD_CURRENT_ADDRESS_TOO_LONG_DESCRIPTION);
        validateOneHundredCharacter(request.getGuardIdPlace(), ErrorCode.GUARD_ID_PLACE_TOO_LONG, ErrorCode.GUARD_ID_PLACE_TOO_LONG_DESCRIPTION);

    }

    private void validateLoanCar(LoanRegistrationExtraRequest request) {
        validateNotBlank(request.getVehicleName(), ErrorCode.VEHICLE_NAME_BLANK, ErrorCode.VEHICLE_NAME_BLANK_DESCRIPTION);
        validateNotBlank(request.getChassisNumber(), ErrorCode.CHASSIS_NUMBER_BLANK, ErrorCode.CHASSIS_NUMBER_BLANK_DESCRIPTION);
        validateNotBlank(request.getEngineNumber(), ErrorCode.ENGINE_NUMBER_BLANK, ErrorCode.ENGINE_NUMBER_BLANK_DESCRIPTION);
        validateNotBlank(request.getVehicleNumber(), ErrorCode.VEHICLE_NUMBER_BLANK, ErrorCode.VEHICLE_NUMBER_BLANK_DESCRIPTION);
        validateNotBlank(request.getVehicleType(), ErrorCode.VEHICLE_TYPE_BLANK, ErrorCode.VEHICLE_TYPE_BLANK_DESCRIPTION);
        validateNotBlank(request.getVehicleBrand(), ErrorCode.VEHICLE_BRAND_BLANK, ErrorCode.VEHICLE_BRAND_BLANK_DESCRIPTION);
        validateNotBlank(request.getVehicleRegistrationIssued(), ErrorCode.VEHICLE_REGISTRATION_ISSUED_BLANK, ErrorCode.VEHICLE_REGISTRATION_ISSUED_BLANK_DESCRIPTION);
        validateNotBlank(request.getVehicleRegistrationDate(), ErrorCode.VEHICLE_REGISTRATION_DATE_BLANK, ErrorCode.VEHICLE_REGISTRATION_DATE_BLANK_DESCRIPTION);
        validateNotBlank(request.getRegistrationNumber(), ErrorCode.REGISTER_NUMBER_BLANK, ErrorCode.REGISTER_NUMBER_BLANK_DESCRIPTION);
        validateNotBlank(request.getFuel(), ErrorCode.FUEL_BLANK, ErrorCode.FUEL_BLANK_DESCRIPTION);
        validateNotBlank(request.getVehicleId(), ErrorCode.VEHICLE_ID_BLANK, ErrorCode.VEHICLE_ID_BLANK_DESCRIPTION);
        validateNotBlank(request.getVehicleOrigin(), ErrorCode.VEHICLE_ORIGIN_BLANK, ErrorCode.VEHICLE_ORIGIN_BLANK_DESCRIPTION);
        validateNotBlank(request.getVehicleColor(), ErrorCode.VEHICLE_COLOR_BLANK, ErrorCode.VEHICLE_COLOR_BLANK_DESCRIPTION);
        validateNotBlank(request.getAddressVehicleDocuments(), ErrorCode.ADDRESS_VEHICLE_DOCUMENT_BLANK, ErrorCode.ADDRESS_VEHICLE_DOCUMENT_BLANK_DESCRIPTION);
        validateNotBlank(request.getNumberPeopleAllow(), ErrorCode.NUMBER_PEOPLE_ALLOW_BLANK, ErrorCode.NUMBER_PEOPLE_ALLOW_BLANK_DESCRIPTION);
        validateSpace(request.getRegSeriNumber(), ErrorCode.REGISTRATION_SERI_NUMBER_SPACE, ErrorCode.REGISTRATION_SERI_NUMBER_SPACE_DESCRIPTION);
        validateSpace(request.getVehicleName(), ErrorCode.VEHICLE_NAME_SPACE, ErrorCode.VEHICLE_NAME_SPACE_DESCRIPTION);
        validateNotBlankInteger(request.getVehYearManufacture(), ErrorCode.YEAR_MANUFACTURE_BLANK, ErrorCode.YEAR_MANUFACTURE_BLANK_DESCRIPTION);
        validateNotBlankInteger(request.getMarket(), ErrorCode.MARKET_BLANK, ErrorCode.MARKET_BLANK_DESCRIPTION);
        validateNotBlankInteger(request.getCommercial(), ErrorCode.COMMERCIAL_BLANK, ErrorCode.COMMERCIAL_BLANK_DESCRIPTION);
        validateNotBlankInteger(request.getVehUseFor(), ErrorCode.USE_FOR_BLANK, ErrorCode.USE_FOR_BLANK_DESCRIPTION);

        DateTimeFormatter formatter = DateTimeFormatter.ofPattern("dd/MM/uuuu").withResolverStyle(ResolverStyle.STRICT);
        if (request.getRegPeriod() != null && !request.getRegPeriod().trim().isEmpty()) {

            try {
                LocalDate.parse(request.getRegPeriod(), formatter);
            } catch (DateTimeParseException e) {
                throw new BusinessException(ErrorCode.REG_PERIOD_FORMAT, ErrorCode.REG_PERIOD_FORMAT_DESCRIPTION);
            }
        }

        if (request.getRegPeriod() != null && !request.getRegPeriod().trim().isEmpty()) {
            if (!request.getRegPeriod().matches("^\\d{2}/\\d{2}/\\d{4}$")) {
                throw new BusinessException(ErrorCode.VEHICLE_REGPERIOD_DATE_FORMAT, ErrorCode.VEHICLE_REGPERIOD_DATE_FORMAT_DESCRIPTION);
            }

            try {
                LocalDate.parse(request.getRegPeriod(), formatter);
            } catch (DateTimeParseException e) {
                throw new BusinessException(ErrorCode.VEHICLE_REGPERIOD_DATE_FORMAT, ErrorCode.VEHICLE_REGPERIOD_DATE_FORMAT_DESCRIPTION);
            }
        }

        if (request.getRegPeriod() != null && !request.getRegPeriod().trim().isEmpty()) {
            LocalDate date;
            try {
                date = LocalDate.parse(request.getRegPeriod().trim(), formatter);

            } catch (DateTimeParseException ex) {
                throw new BusinessException(ErrorCode.VEHICLE_REGPERIOD_DATE_FORMAT, ErrorCode.VEHICLE_REGPERIOD_DATE_FORMAT_DESCRIPTION);
            }

//        if (!date.isBefore(LocalDate.now())) {
//            throw new BusinessException(errorCode, errorDesc);
//        }

            if (date.getYear() < 1900) {
                throw new BusinessException(ErrorCode.VEHICLE_REGPERIOD_DATE_FORMAT, ErrorCode.VEHICLE_REGPERIOD_DATE_FORMAT_DESCRIPTION);
            }
        }

        if (request.getLoanPrice() == null) {
            throw new BusinessException(ErrorCode.LOAN_PRICE_BLANK, ErrorCode.LOAN_PRICE_BLANK_DESCRIPTION);
        }

        if (request.getVehicleName().trim().length() > 100 || !request.getVehicleName().matches("^[\\p{L}\\s]+$")) {
            throw new BusinessException(ErrorCode.VEHICLE_NAME_TOO_LONG, ErrorCode.VEHICLE_NAME_TOO_LONG_DESCRIPTION);
        }

        String seri = request.getRegSeriNumber();
        if (seri != null && !seri.trim().isEmpty()) {
            seri = seri.trim();
            if (seri.length() > 20 || !seri.matches("^[a-zA-Z0-9]+$")) {
                throw new BusinessException(
                        ErrorCode.REGISTRATION_SERI_NUMBER_TOO_LONG,
                        ErrorCode.REGISTRATION_SERI_NUMBER_TOO_LONG_DESCRIPTION
                );
            }
        }

        if (request.getVehYearManufacture() < 1900 || request.getVehYearManufacture() > 2025) {
            throw new BusinessException(ErrorCode.YEAR_MANUFACTURE_VALID, ErrorCode.YEAR_MANUFACTURE_VALID_DESCRIPTION);
        }


//        if (request.getSanction() < 0 || request.getSanction() > 10000000000) {
//            throw new BusinessException(ErrorCode.SANCTION_VALID, ErrorCode.SANCTION_VALID_DESCRIPTION);
//        }

        if (request.getSanction() != null) {
            BigDecimal zero = BigDecimal.ZERO;
            BigDecimal tenBillion = new BigDecimal("10000000000");

            if (request.getSanction().compareTo(zero) < 0 ||
                    request.getSanction().compareTo(tenBillion) > 0) {
                throw new BusinessException(ErrorCode.SANCTION_VALID, ErrorCode.SANCTION_VALID_DESCRIPTION);
            }
        }

        if (request.getVehiclePrice() != null) {
            if (request.getVehiclePrice().compareTo(BigDecimal.ZERO) < 0 || request.getVehiclePrice().compareTo(new BigDecimal("1000000000000")) > 0) {
                throw new BusinessException(ErrorCode.VEHICLE_PRICE_VALID, ErrorCode.VEHICLE_PRICE_VALID_DESCRIPTION);
            }
        }

        if (request.getLoanPrice().compareTo(BigDecimal.ZERO) < 0 || request.getLoanPrice().compareTo(new BigDecimal("1000000000000")) > 0) {
            throw new BusinessException(ErrorCode.LOAN_PRICE_VALID, ErrorCode.LOAN_PRICE_VALID_DESCRIPTION);
        }

        if (!request.getVehicleColor().matches("^[\\p{L} ]+$")) {
            throw new BusinessException(ErrorCode.ERROR, ErrorCode.VEHICLE_COLOR_NUMBER_DESCRIPTION);
        }

        if (!request.getNumberPeopleAllow().matches("^[0-9]+$")) {
            throw new BusinessException(ErrorCode.ERROR, ErrorCode.NUMBER_PEOPLE_ALLOW);
        }
        if (request.getNumberPeopleAllow().length() > 20) {
            throw new BusinessException(ErrorCode.ERROR, ErrorCode.NUMBER_PEOPLE_ALLOW_MAX);
        }

        DateValidator.validateDateFormat(request.getVehicleRegistrationDate(), ErrorCode.VEHICLE_REGISTRATION_DATE_FORMAT, ErrorCode.VEHICLE_REGISTRATION_DATE_FORMAT_DESCRIPTION);
        DateValidator.validateRealDate(request.getVehicleRegistrationDate(), ErrorCode.VEHICLE_REGISTRATION_DATE_REAL, ErrorCode.VEHICLE_REGISTRATION_DATE_REAL_DESCRIPTION);
        DateValidator.validateRealDateNow(request.getVehicleRegistrationDate(), ErrorCode.VEHICLE_REGISTRATION_DATE_NOW, ErrorCode.VEHICLE_REGISTRATION_DATE_FORMAT_DESCRIPTION_NOW);

//        DateValidator.validateDateFormat(request.getRegPeriod(), ErrorCode.VEHICLE_REGPERIOD_DATE_FORMAT, ErrorCode.VEHICLE_REGPERIOD_DATE_FORMAT_DESCRIPTION);
//        DateValidator.validateRealDate(request.getRegPeriod(), ErrorCode.VEHICLE_REGPERIOD_DATE_REAL, ErrorCode.VEHICLE_REGPERIOD_DATE_REAL_DESCRIPTION);

        validateTwentyCharacter(request.getRegLoad(), ErrorCode.ASSET_REG_LOAD_TOO_LONG, ErrorCode.ASSET_REG_LOAD_TOO_LONG_DESCRIPTION);
        validateFiftyCharacter(request.getVehicleNumber(), ErrorCode.VEHICLE_NUMBER_TOO_LONG, ErrorCode.VEHICLE_NUMBER_TOO_LONG_DESCRIPTION);
        validateFiftyCharacter(request.getRegistrationNumber(), ErrorCode.REGISTRATION_NUMBER_TOO_LONG, ErrorCode.REGISTRATION_NUMBER_TOO_LONG_DESCRIPTION);
        validateTwentyCharacter(request.getRegSeriNumber(), ErrorCode.ASSET_REG_SERI_NUMBER_TOO_LONG, ErrorCode.ASSET_REG_SERI_NUMBER_TOO_LONG_DESCRIPTION);
        validateFiftyCharacter(request.getChassisNumber(), ErrorCode.CHASSIS_NUMBER_TOO_LONG, ErrorCode.CHASSIS_NUMBER_TOO_LONG_DESCRIPTION);
        validateFiftyCharacter(request.getEngineNumber(), ErrorCode.ENGINE_NUMBER_TOO_LONG, ErrorCode.ENGINE_NUMBER_TOO_LONG_DESCRIPTION);
        validateTwentyCharacter(request.getFuel(), ErrorCode.FUEL_TOO_LONG, ErrorCode.FUEL_TOO_LONG_DESCRIPTION);
        validateTwentyCharacter(request.getVehicleOrigin(), ErrorCode.VEHICLE_ORIGIN_TOO_LONG, ErrorCode.VEHICLE_ORIGIN_TOO_LONG_DESCRIPTION);
        validateTwentyCharacter(request.getVehicleColor(), ErrorCode.VEHICLE_COLOR_TOO_LONG, ErrorCode.VEHICLE_COLOR_TOO_LONG_DESCRIPTION);
        validateTwentyCharacter(request.getTransPor(), ErrorCode.TRANSPOR_TOO_LONG, ErrorCode.TRANSPOR_TOO_LONG_DESCRIPTION);
        validateTwentyCharacter(request.getRenovation(), ErrorCode.RENOVATION_TOO_LONG, ErrorCode.RENOVATION_TOO_LONG_DESCRIPTION);
        validateTwentyCharacter(request.getVehicleInspectionModelCode(), ErrorCode.MODEL_CODE_TOO_LONG, ErrorCode.MODEL_CODE_TOO_LONG_DESCRIPTION);
        validateOneHundredCharacter(request.getAddressVehicleDocuments(), ErrorCode.ADDRESS_VEHICLE_DOCUMENT_TOO_LONG, ErrorCode.ADDRESS_VEHICLE_DOCUMENT_TOO_LONG_DESCRIPTION);
        validateOneHundredCharacter(request.getVehicleRegistrationIssued(), ErrorCode.IDDATE_CAR_DOCUMENT_TOO_LONG, ErrorCode.AIDDATE_CAR_DOCUMENT_TOO_LONG_DESCRIPTION);
//        validateTenBillion(request.getSanction(), ErrorCode.ERROR, ErrorCode.SANCTION_CAR_TENBILLION);

        DateValidator.validateDateFormatAndExistence(request.getVehicleRegistrationDate(),
                ErrorCode.BUSINESS_OWNER_UPDATE_DOB_NOT_REAL,
                ErrorCode.BUSINESS_OWNER_UPDATE_DOB_NOT_REAL_DESCRIPTION,
                ErrorCode.VEHICLE_RES_DATE_WRONG_FORMAT,
                ErrorCode.VEHICLE_RES_DATE_WRONG_FORMAT_DESCRIPTION);

        ValidUtils.validateMaxLength(request.getAddressVehicleDocuments(),200,
                ErrorCode.VEHICLE_ADDRESS_UPDATE_OVER150,
                ErrorCode.VEHICLE_ADDRESS_UPDATE_OVER150_DESCRIPTION);

        ValidUtils.validateMaxLength(request.getVehicleRegistrationIssued(),100,
                ErrorCode.VEHICLE_REGISTRATION_UPDATE_OVER100,
                ErrorCode.VEHICLE_REGISTRATION_UPDATE_OVER100_DESCRIPTION);

    }

    private void validateLoanCarList(List<LoanRegistrationExtraRequest> list) {
        if (list == null || list.isEmpty()) {
            throw new BusinessException(ErrorCode.CAR_LIST_EMPTY, ErrorCode.CAR_LIST_EMPTY_DESCRIPTION);

        }
        for (LoanRegistrationExtraRequest loanRegistrationExtraRequest : list) {
            validateLoanCar(loanRegistrationExtraRequest);
        }
    }
